Let’s use bill_length_mm to predict bill_depth_mm. Prepare your data and fit the following models on the entire dataset:

  • Simple linear regression (e.g. straight-line) model
  • Quadratic (degree 2 polynomial) model
  • Cubic (degree 3 polynomial) model
  • Degree 10 polynomial model

Make predictions for each model and plot your fitted models on the scatterplot.

from sklearn.preprocessing import OneHotEncoder, StandardScaler
import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression
from sklearn.preprocessing import PolynomialFeatures
from plotnine import *

# Standard Scaling for numerical features
scaler = StandardScaler()
numerical_features = ['bill_length_mm', 'bill_depth_mm', 'flipper_length_mm', 'body_mass_g']
penguins[numerical_features] = scaler.fit_transform(penguins[numerical_features])

# Prepare the data
X = penguins[['bill_length_mm']]
y = penguins['bill_depth_mm']

# Fit a simple linear regression model
linear_model = LinearRegression()
linear_model.fit(X, y)
penguins['predictions_linear'] = linear_model.predict(X) # adding prediction column

# Create a scatterplot with plotnine
(ggplot(penguins, aes(x='bill_length_mm', y='bill_depth_mm'))
 + geom_point()
 + geom_line(aes(y='predictions_linear'), color='red')
 + labs(title='Bill Length vs. Bill Depth with Linear Model', x='Bill Length (mm)', y='Bill Depth (mm)')
)


# Fit a quadratic model
poly_features_2 = PolynomialFeatures(degree=2)
X_poly_2 = poly_features_2.fit_transform(X)
quadratic_model = LinearRegression()
quadratic_model.fit(X_poly_2, y)
penguins['predictions_quadratic'] = quadratic_model.predict(X_poly_2)

# Create a scatterplot with plotnine for quadratic fit
(ggplot(penguins, aes(x='bill_length_mm', y='bill_depth_mm'))
 + geom_point()
 + geom_line(aes(y='predictions_quadratic'), color='green')
 + labs(title='Bill Length vs. Bill Depth with Quadratic Model', x='Bill Length (mm)', y='Bill Depth (mm)')
)
# Fit a cubic model
poly_features_3 = PolynomialFeatures(degree=3)
X_poly_3 = poly_features_3.fit_transform(X)
cubic_model = LinearRegression()
cubic_model.fit(X_poly_3, y)
penguins['predictions_cubic'] = cubic_model.predict(X_poly_3)

# Create a scatterplot with plotnine for cubic fit
(ggplot(penguins, aes(x='bill_length_mm', y='bill_depth_mm'))
 + geom_point()
 + geom_line(aes(y='predictions_cubic'), color='blue')
 + labs(title='Bill Length vs. Bill Depth with Cubic Model', x='Bill Length (mm)', y='Bill Depth (mm)')
)




# Fit a degree 10 polynomial model
poly_features_10 = PolynomialFeatures(degree=10)
X_poly_10 = poly_features_10.fit_transform(X)
degree_10_model = LinearRegression()
degree_10_model.fit(X_poly_10, y)
penguins['predictions_degree_10'] = degree_10_model.predict(X_poly_10)

# Create a scatterplot with plotnine for degree 10 polynomial fit
(ggplot(penguins, aes(x='bill_length_mm', y='bill_depth_mm'))
 + geom_point()
 + geom_line(aes(y='predictions_degree_10'), color='orange')
 + labs(title='Bill Length vs. Bill Depth with Degree 10 Polynomial Model', x='Bill Length (mm)', y='Bill Depth (mm)')
)
from sklearn.metrics import mean_squared_error

# ... (previous code for imports, data preprocessing, and model fitting) ...

# Calculate MSE for each model
mse_linear = mean_squared_error(y, penguins['predictions_linear'])
mse_quadratic = mean_squared_error(y, penguins['predictions_quadratic'])
mse_cubic = mean_squared_error(y, penguins['predictions_cubic'])
mse_degree_10 = mean_squared_error(y, penguins['predictions_degree_10'])

# Print the MSE values
print(f"Linear Regression MSE: {mse_linear}")
print(f"Quadratic Regression MSE: {mse_quadratic}")
print(f"Cubic Regression MSE: {mse_cubic}")
print(f"Degree 10 Polynomial Regression MSE: {mse_degree_10}")

# Find the model with the lowest MSE
min_mse = min(mse_linear, mse_quadratic, mse_cubic, mse_degree_10)

if min_mse == mse_linear:
    best_model = "Linear Regression"
elif min_mse == mse_quadratic:
    best_model = "Quadratic Regression"
elif min_mse == mse_cubic:
    best_model = "Cubic Regression"
else:
    best_model = "Degree 10 Polynomial Regression"

print(f"\nThe best-fitting model is: {best_model}")

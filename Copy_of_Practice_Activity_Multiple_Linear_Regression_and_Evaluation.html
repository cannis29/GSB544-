# Import necessary libraries
from palmerpenguins import load_penguins
import pandas as pd
from sklearn.pipeline import Pipeline
from sklearn.linear_model import LinearRegression
from sklearn.preprocessing import OneHotEncoder, PolynomialFeatures
from sklearn.model_selection import train_test_split
from sklearn.compose import ColumnTransformer
from sklearn.metrics import r2_score, mean_squared_error

# Load the penguins dataset
penguins = load_penguins()

# Display the first few rows of the dataset
print(penguins.head())

# Drop rows with any missing values
penguins = penguins.dropna()

# Separate features (X) and target variable (y)
X = penguins.drop('bill_depth_mm', axis=1)
y = penguins['bill_depth_mm']

# Split the data into training and testing sets
X_train, X_test, y_train, y_test = train_test_split(X, y, random_state=0)

# Preprocessing step: OneHotEncode categorical variables
# Model specification: Linear Regression
lr = LinearRegression()
enc = OneHotEncoder()

# Column transformer to apply OneHotEncoder to categorical features
ct = ColumnTransformer([('dummify', enc, ['species', 'island', 'sex', 'year'])], remainder='passthrough')

# Pipeline setup with preprocessing and model steps
my_pipeline_lr = Pipeline([
    ('dummify_everything', ct),  # Apply the column transformer
    ('ols', lr)                  # Apply Linear Regression
])

# Fit the pipeline on the training data
fitted_pipeline_lr = my_pipeline_lr.fit(X_train, y_train)

# Make predictions on both training and testing sets
y_train_hat = fitted_pipeline_lr.predict(X_train)
y_test_hat = fitted_pipeline_lr.predict(X_test)

# Calculate evaluation metrics
r2_train = r2_score(y_train, y_train_hat)  # R-squared on training data
mse = mean_squared_error(y_test, y_test_hat)  # Mean Squared Error on test data
r2 = r2_score(y_test, y_test_hat)  # R-squared on test data
num_coeffs = fitted_pipeline_lr.named_steps['ols'].coef_.shape[0]  # Number of coefficients in the model

# Output the results
print(f"Training R-squared: {r2_train}")
print(f"Test Mean Squared Error: {mse}")
print(f"Test R-squared: {r2}")
print(f"Number of coefficients: {num_coeffs}")
